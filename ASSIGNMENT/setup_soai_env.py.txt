import subprocess
import sys
import shutil

def run_command(command, check=True):
    print(f"ğŸ‘‰ Running: {command}")
    result = subprocess.run(command, shell=True)
    if check and result.returncode != 0:
        print(f"âŒ Command failed: {command}")
        sys.exit(1)
    return result.returncode == 0

def check_git():
    print("ğŸ” Checking Git...")
    result = subprocess.run("git --version", capture_output=True, text=True)
    if "git version" in result.stdout:
        version = result.stdout.strip().split()[-1]
        print(f"âœ… Git installed: {version}")
    else:
        print("âŒ Git not found. Please install Git from https://git-scm.com/")
        sys.exit(1)

def check_python():
    print("ğŸ Checking Python version...")
    version = sys.version_info
    if version.major == 3 and version.minor >= 11:
        print(f"âœ… Python version OK: {version.major}.{version.minor}")
    else:
        print("âŒ Python 3.11+ required.")
        sys.exit(1)

def check_uv():

def is_installed(command):
    return shutil.which(command) is not None

def install_vs_code():
    if is_installed("code"):
        print("âœ… VS Code already installed.")
    else:
        print("â¬‡ï¸ Installing VS Code using winget...")
        run_command('winget install --id Microsoft.VisualStudioCode -e --source winget')

def install_python_packages():
    packages = ["numpy", "pandas", "streamlit", "jupyterlab", "ruff"]
    for pkg in packages:
        print(f"ğŸ“¦ Installing {pkg} via uv")
        run_command(f"uv pip install --upgrade {pkg} --system")
